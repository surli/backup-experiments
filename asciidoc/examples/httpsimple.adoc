----
package org.rapidoid.docs.httpsimple;

import org.rapidoid.setup.On;

public class Main {

	public static void main(String[] args) {
		/* On [GET /size] return the length of the "msg" parameter */

		On.get("/size").json((String msg) -> msg.length());
	}

}
