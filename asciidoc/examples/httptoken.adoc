### Client-side Token as a session

[[app-listing]]
[source,java]
.Main.java
----
package org.rapidoid.docs.httptoken;

import org.rapidoid.setup.On;

public class Main {

	public static void main(String[] args) {
		On.req(req -> {
			int counter = req.token("n", 0) + 1;
			req.token().put("n", counter);
			return counter;
		});
	}

}
----
